DATABASE turret

    User accounts
    TABLE accounts
        KEY   userid   INT
        
        FIELD time     INT
        FIELD username STR
        FIELD password STR

    unique errors
    TABLE issues
        KEY   errid    INT          
        
        FIELD time     INT             timestamp accurate to seconds
        FIELD content  STR 
        
    unique events
    TABLE events
        KEY   eventid  INT             separate in order to keep events unique from errors

        KEY   errid    INT             
        FIELD time     INT             timestamp accurate to days
        FIELD userid   INT
        

    create table issues( 
        issueid INT NOT NULL AUTO_INCREMENT, 
        time TIMESTAMP NOT NULL, 
        content VARCHAR(256) NOT NULL, 
        
        PRIMARY KEY (issueid)
    );

    create table events(
        eventid INT NOT NULL AUTO_INCREMENT,
        issueid   INT NOT NULL,

        time TIMESTAMP NOT NULL,

        PRIMARY KEY (eventid),
        FOREIGN KEY (issueid) REFERENCES issues(issueid)
    );
